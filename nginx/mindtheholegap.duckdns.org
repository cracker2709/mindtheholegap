server {
    listen 443 ssl;
    server_name mindtheholegap.duckdns.org www.mindtheholegap.duckdns.org;

    ssl_certificate /etc/letsencrypt/live/mindtheholegap.duckdns.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/mindtheholegap.duckdns.org/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    root /var/www/html;
    index index.html;

    location /.well-known/acme-challenge/ {
        root /var/www/html;
    }

    # Block access to sensitive files and directories
    location ~ /\.git {
        deny all;
        return 404;
    }

    location ~ /\. {
        deny all;
        return 404;
    }

    # Block access to backup and source files
    location ~ ~$ {
        deny all;
    }

    location ~ \.(bak|config|sql|fla|psd|ini|log|sh|inc|swp|dist|env.*|yml|yaml)$ {
        deny all;
        return 404;
    }

         # Try URI as-is, then with .html, then as a directory
        try_files $uri $uri.html $uri/ =404;
    }
    
    location /compteur {
        content_by_lua_block {
        local visites = ngx.shared.visites
        local ip = ngx.var.remote_addr

        -- si l'IP n'a jamais visité, on compte
        if visites:get(ip) == nil then
            visites:set(ip, true)

            local total = visites:get("total") or 0
            total = total + 1
            visites:set("total", total)
        end

        ngx.say(visites:get("total") or 1)
        }
    }


    location /assets/ {
        # Serve static files directly
        try_files $uri =404;
    }
}

server {
    listen 80;
    server_name mindtheholegap.duckdns.org www.mindtheholegap.duckdns.org;

    location / {
        return 301 https://$host:8443$request_uri;
    }
}
